            e.preventDefault();

            const from = document.getElementById('fromCity').value;
            const to = document.getElementById('toCity').value;
            const depart = document.getElementById('departDate').value;
            const returnDate = document.getElementById('returnDate').value;
            const tripType = document.querySelector('input[name="tripType"]:checked').value;
            const cabinClass = document.getElementById('cabinClass').value;

            const params = new URLSearchParams({
                from: from,
                to: to,
                depart: depart,
                type: tripType,
                adults: travelersData.adults,
                children: travelersData.children,
                infants: travelersData.infants,
                class: cabinClass
            });

            // One Way trip type check
            if (tripType === 'oneway' && returnDate) {
                // Don't append return date
            } else if (tripType === 'roundtrip' && returnDate) {
                params.append('return', returnDate);
            }

            window.location.href = `flight-results.html?${params.toString()}`;
        });

        // ========== Signup Promo Modal Logic ==========
        document.addEventListener('DOMContentLoaded', async () => {
            // Delay check to ensure Supabase client is ready
            setTimeout(async () => {
                let isLoggedIn = false;

                // Check if SupabaseDB is defined (from supabase-client.js)
                if (typeof SupabaseDB !== 'undefined') {
                    try {
                        const user = await SupabaseDB.getCurrentUser();
                        if (user) isLoggedIn = true;
                    } catch (e) {
                        console.log('User check failed', e);
                    }
                }

                // If not logged in, maintainer simple logic
                if (!isLoggedIn) {
                    // Check if dismissed recently (24h)
                    const dismissed = localStorage.getItem('signupModalDismissed');
                    const now = new Date().getTime();

                    if (!dismissed || (now - parseInt(dismissed) > 24 * 60 * 60 * 1000)) {
                        // Show after 10 seconds of browsing
                        setTimeout(() => {
                            document.getElementById('signupModal').classList.add('active');
                        }, 10000);
                    }
                }
            }, 1000);

            // Close handlers
            const closeSignup = () => {
                document.getElementById('signupModal').classList.remove('active');
                localStorage.setItem('signupModalDismissed', new Date().getTime().toString());
            };

            const closeBtn = document.getElementById('signupClose');
            const laterBtn = document.getElementById('signupLater');
            const modalOverlay = document.getElementById('signupModal');

            if (closeBtn) closeBtn.addEventListener('click', closeSignup);
            if (laterBtn) laterBtn.addEventListener('click', closeSignup);
            if (modalOverlay) {
                modalOverlay.addEventListener('click', (e) => {
                    if (e.target === modalOverlay) closeSignup();
                });
            }
        });
